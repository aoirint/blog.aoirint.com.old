Download a file

file_url: str
dest_path: str

with tempfile.NamedTemporaryFile() as fp:
    with ses.get(file_url, stream=True) as r:
        r.raise_for_status()

        for chunk in r.iter_content(chunk_size=8192):
            fp.write(chunk)

    fp.flush()
    shutil.copy(fp.name, dest_path)
